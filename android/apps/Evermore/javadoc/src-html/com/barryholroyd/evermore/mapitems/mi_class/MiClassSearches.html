013import com.barryholroyd.evermore.mapitems.mapitem.MapItem;
014import com.barryholroyd.evermore.mapitems.mapitem.MapItemRegistry;
015import com.barryholroyd.evermore.mapitems.mi_subclass.*;
016import com.barryholroyd.evermore.mapitems.mi_type.MiType;
017import com.barryholroyd.evermore.yelp.dialogs.YelpMarkerInfoWindowDf;
018import com.google.android.gms.maps.model.Marker;
019
020public class MiClassSearches extends MiClass {
021        private static MiClassSearches miClassSearches = null;
022
023        private MiClassSearches(MiType p) {
024                this.parent = p;
025                addMiSubclass(MiSubclassSearch1.create(this));
026                addMiSubclass(MiSubclassSearch2.create(this));
027                addMiSubclass(MiSubclassSearch3.create(this));
028                addMiSubclass(MiSubclassSearch4.create(this));
029                addMiSubclass(MiSubclassSearch5.create(this));
030                addMiSubclass(MiSubclassSearch6.create(this));
031                addMiSubclass(MiSubclassSearch7.create(this));
032        }
033        static public MiClassSearches create(MiType p) {
034                if (miClassSearches == null)
035                        miClassSearches = new MiClassSearches(p);
036                return miClassSearches;
037        }
038        static public MiClassSearches i() { return miClassSearches; }
039        public void markerInfoWindowClicked(Marker marker) {
040                MapItem          mapItem          = MapItemRegistry.getMapItemUsingMarker(marker);
041                MiSubclassSearch miSubclassSearch = (MiSubclassSearch) mapItem.getMiSubclass();
042
043                switch (miSubclassSearch.getSearchType()) {
044                        case Yelp:
045                                YelpMarkerInfoWindowDf.showMe(mapItem);
046                                break;
047                }
048                super.markerInfoWindowClicked(marker);
049        }
050}
