013import com.barryholroyd.evermore.mapitems.mapitem.MapItemHashSet;
014import com.barryholroyd.evermore.mapitems.mi_type.*;
015
016public class MiTop
017{
018        //<editor-fold desc="CONSTRUCTOR, SINGLETON">
019        private static MiTop miTop = null;
020        private MiTop() {
021                addMiType(MiTypeTargets.create(this));
022                addMiType(MiTypePins.create(this));
023        }
024        static public void create() {
025                if (miTop == null)
026                        miTop = new MiTop();
027        }
028        @SuppressWarnings("unused")
029        static public MiTop i() { return miTop; }
030//</editor-fold>
031
032        //<editor-fold desc="STATIC FIELDS AND METHODS">
033        private static final MiTypeArray miTypesAll = new MiTypeArray();
034        private static MiTypeArray getAllMiTypes() {
035                // get all the Subclasses
036                return miTypesAll;
037        }
038        @SuppressWarnings("unused")
039        static public void clearMiTypes() {
040                //noinspection StatementWithEmptyBody
041                for (MiType miType : getAllMiTypes()) {
042                        // not implemented
043                }
044                throw new IllegalStateException("clearMiTypes not implemented");
045        }
046        //</editor-fold>
047
048        private final MiTypeArray miTypes = new MiTypeArray();
049        private void addMiType(MiType miType) {
050                miTypes.add(miType);
051                miTypesAll.add(miType);
052        }
053        @SuppressWarnings("unused")
054        public MapItemHashSet getAllMapItems() {
055                return getAllMyMapItems();
056        }
057        private MapItemHashSet getAllMyMapItems() {
058                MapItemHashSet mapItemsHm = new MapItemHashSet();
059                for (MiType miType : getMiTypes()) {
060                        mapItemsHm.addAll(miType.getAllMyMapItems());
061                }
062                return mapItemsHm;
063        }
064        @SuppressWarnings("unused")
065        public Void getMyParent() { return null; }
066        // Same as getAlLMiTypes(), but included for consistency.
067        private MiTypeArray getMiTypes() { // all miTypes
068                return miTypes;
069        }
070}
