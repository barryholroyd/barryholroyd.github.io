013/*
014 * Some Evermore entities have one or more addresses associated with them.
015 * Currently, those entity types include:
016 *
017 *   1. The owner of the app.
018 *   2. Contacts pulled from the owner's address book.
019 *
020 * In all cases, these address are unique to the thing they apply to, e.g., each
021 * Entity has at most one of each.
022 *
023 * This is used by MiSubclassKeymarks to represent Markers representing the owner's unique
024 * addresses (e.g., Home, Work, Birthplace) and by MiSubclassContacts to represent Contacts'
025 * addresses (e.g., Home, Work).
026 */
027
028import com.barryholroyd.evermore.ActivityMain;
029import com.barryholroyd.evermore.R;
030
031// Used by Keymarks and by Contacts.
032public enum AdrsTypeEnum
033{
034        // We need the rid so that we can get device-dependent strings.
035        NEITHER(R.string.list_neither),
036        HOME(R.string.list_home),
037        WORK(R.string.list_work),
038        BIRTHPLACE(R.string.list_birthplace), // included as an (unlikely) example of a third item
039        ;
040        private final int rid;
041        @SuppressWarnings("unused")
042        AdrsTypeEnum(int _rid) { rid = _rid; }
043
044        public String getLabel() {
045                return ActivityMain.getActivity().getResources().getString(rid);
046        }
047        @SuppressWarnings("unused")
048        static public AdrsTypeEnum getValue(String label) {
049                for (AdrsTypeEnum atEnum : AdrsTypeEnum.values()) {
050                        if (label.equals(atEnum.getLabel()))
051                                return atEnum;
052                }
053                throw new IllegalStateException(
054                        String.format("AdrsTypeEnum label not found: %s", label));
055        }
056}
