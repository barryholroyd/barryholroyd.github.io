013import com.google.android.gms.location.LocationRequest;
014
015import static com.google.android.gms.location.LocationRequest.PRIORITY_BALANCED_POWER_ACCURACY;
016import static com.google.android.gms.location.LocationRequest.PRIORITY_HIGH_ACCURACY;
017import static com.google.android.gms.location.LocationRequest.PRIORITY_NO_POWER;
018
019public enum Quality
020{
021        LOW(PRIORITY_NO_POWER, QualityDefaults.IGNORE, QualityDefaults.IGNORE),
022        @SuppressWarnings("unused")MEDIUM(PRIORITY_BALANCED_POWER_ACCURACY, 5*60, 60),
023        HIGH(PRIORITY_HIGH_ACCURACY, 5, QualityDefaults.IGNORE),
024        ;
025        private final int priority;
026        private final int interval;   // seconds
027        private final int fastest;    // seconds
028
029        @SuppressWarnings("unused")
030        Quality(int _priority, int _interval, int _fastest) {
031                priority = _priority;
032                interval = _interval;
033                fastest = _fastest;
034        }
035
036        LocationRequest getLocationRequest() {
037                LocationRequest lr = new LocationRequest();
038                lr.setPriority(priority);
039                if (interval != QualityDefaults.IGNORE) lr.setInterval(interval * 1000);
040                if (fastest != QualityDefaults.IGNORE) lr.setFastestInterval(fastest * 1000);
041                return lr;
042        }
043}
044
045// Need this because it can't be defined within the Quality enum itself.
046class QualityDefaults { static final int IGNORE = -1; }
