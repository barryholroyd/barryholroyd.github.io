013import com.barryholroyd.evermore.mapitems.MiTop;
014import com.barryholroyd.evermore.mapitems.mi_class.*;
015
016public class MiTypePins extends MiType {
017        private static MiTypePins miTypePins = null;
018        private MiTypePins(MiTop p) {
019                this.parent = p;
020                addMiClass(MiClassContacts.create(this));
021                addMiClass(MiClassKeymarks.create(this));
022                addMiClass(MiClassSearches.create(this));
023        }
024        static public MiTypePins create(MiTop p) {
025                if (miTypePins == null) {
026                        miTypePins = new MiTypePins(p);
027                }
028                return miTypePins;
029        }
030        @SuppressWarnings("unused")
031        static public MiTypePins i() { return miTypePins; }
032}
