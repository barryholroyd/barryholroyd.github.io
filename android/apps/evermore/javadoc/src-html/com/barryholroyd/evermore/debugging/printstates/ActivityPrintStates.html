013import android.os.Bundle;
014import android.support.v7.app.AppCompatActivity;
015
016import com.barryholroyd.evermore.debugging.controlpanel.Tracer;
017import com.barryholroyd.evermore.debugging.controlpanel.CpConfigure.*;
018
019import com.barryholroyd.evermore.debugging.logger.BhLogger;
020
021@SuppressWarnings("unused")
022abstract public class ActivityPrintStates extends AppCompatActivity
023{
024  //<editor-fold desc="FIELDS">
025  private final BhLogger bh = new BhLogger();
026  private boolean extendedLogTag = false;
027  //</editor-fold>
028  //<editor-fold desc="OVERRIDDEN METHODS">
029  // Called at the start of the full lifetime.
030  @Override
031  protected void onCreate(Bundle savedInstanceState) {
032    super.onCreate(savedInstanceState);
033    // Initialize Activity and inflate the UI.
034    runme("onCreate()");
035  }
036
037  // Called after onCreate has finished, use to restore UI state
038  @Override
039  protected void onRestoreInstanceState(Bundle savedInstanceState) {
040    super.onRestoreInstanceState(savedInstanceState);
041    // Restore UI state from the savedInstanceState.
042    // This bundle has also been passed to onCreate.
043    // Will only be called if the Activity has been 
044    // killed by the system since it was last visible.
045    runme("onRestoreInstanceState()");
046  }
047
048  // Called before subsequent visible lifetimes
049  // for an activity process.
050  @Override
051  protected void onRestart(){
052    super.onRestart();
053    // Load changes knowing that the Activity has already
054    // been visible within this process.
055    runme("onRestart()");
056  }
057
058  // Called at the start of the visible lifetime.
059  @Override
060  protected void onStart(){
061    super.onStart();
062    // Apply any required UI change now that the Activity is visible.
063    runme("onStart()");
064  }
065
066  // Called at the start of the active lifetime.
067  @Override
068  protected void onResume(){
069    super.onResume();
070    // Resume any paused UI updates, threads, or processes required
071    // by the Activity but suspended when it was inactive.
072    runme("onResume()");
073  }
074
075  // Called to save UI state changes at the
076  // end of the active lifecycle.
077  @Override
078  protected void onSaveInstanceState(Bundle savedInstanceState) {
079    // Save UI state changes to the savedInstanceState.
080    // This bundle will be passed to onCreate and 
081    // onRestoreInstanceState if the process is
082    // killed and restarted by the run time.
083    super.onSaveInstanceState(savedInstanceState);
084    runme("onSaveInstanceState()");
085  }
086
087  // Called at the end of the active lifetime.
088  @Override
089  protected void onPause(){
090    // Suspend UI updates, threads, or CPU intensive processes
091    // that don't need to be updated when the Activity isn't
092    // the active foreground Activity.
093    super.onPause();
094    runme("onPause()");
095  }
096
097  // Called at the end of the visible lifetime.
098  @Override
099  protected void onStop(){
100    // Suspend remaining UI updates, threads, or processing
101    // that aren't required when the Activity isn't visible.
102    // Persist all edits or state changes
103    // as after this call the process is likely to be killed.
104    super.onStop();
105    runme("onStop()");
106  }
107
108  // Sometimes called at the end of the full lifetime.
109  @Override
110  protected void onDestroy(){
111    // Clean up any resources including ending threads,
112    // closing database connections etc.
113    super.onDestroy();
114    runme("onDestroy()");
115  }
116  //</editor-fold>
117  //<editor-fold desc="SUPPORT METHODS">
118  @SuppressWarnings("unused")
119  protected void extendLogtag(String s) {
120    // Note: the call from the subclass to this method should be done from the subclass'
121    // onAttach() method so that the tag kicks in immediately in the fragment's lifecycle.
122    if (!extendedLogTag) {
123      extendedLogTag = true;
124      bh.extendLogtag(s);
125    }
126  }
127  private void runme(String label) {
128    Tracer.log(bh, CpControl.LC_ACTIVITY, label);
129  }
130  //</editor-fold>
131}
