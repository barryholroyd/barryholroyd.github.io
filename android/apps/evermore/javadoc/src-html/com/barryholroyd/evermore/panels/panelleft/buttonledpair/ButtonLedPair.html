034/*
035 *************************************************************************************************
036 * OVERVIEW
037 *************************************************************************************************
038 *   ButtonLedPair is a compound widget that pairs a search button with with an LED button
039 *   so that the pressing of the search button toggles the LED, as appropriate. Most
040 *   initialization, including hooking together the two buttons as well as adding
041 *   callbacks for them, is done in the constructor.
042 * 
043 *   In addition, an MiSubclass instance is identified and hooked in, although that initialization
044 *   must be a "late initialization" since it has to wait for the MiSubclass instances to
045 *   be created.
046 *
047 *   The ButtonLedPair and MiSubclass instances have pointers to each other, with ButtonLedPair
048 *   containing the static linkage information and MiSubclass containing the dynamic information.
049 *************************************************************************************************
050 * FLOW OF CONTROL
051 *************************************************************************************************
052 *   Initialization
053 *     o App start: left panel is laid out with ButtonLedPairs. This class's constructor is called
054 *       for each ButtonLedPair as part of the UI building process.
055 *     o Constructor calls cinit():
056 *       o Inflates pair_searchbutton which inflates a ButtonSearch and a ButtonLed.
057 *       o Initializes each ButtonSearch (label (e.g., "Search 1"), callbacks).
058 *     o PanelLeft.init() calls lateinit(), which hooks everything together. Late initialization
059 *       has to happen because each ButtonLedPair needs to get hooked up with one of the (unique)
060 *       MiSubclassSearch* instances and those haven't been created yet when the UI is initially
061 *       laid out. The following all get hooked together.
062 *       o ButtonLedPair
063 *       o MiSubclassSearch* instance
064 *       o ButtonSearch
065 *       o ButtonLed
066 *       o ControlButtonListeners (ocl)
067 *     Usage
068 *       o The ButtonLedPair and MiSubclassSearch instances point at each other
069 *         and form the root of the tree from which the rest of the components
070 *         can be accessed.
071 *           o ButtonLedPair from button clicks.
072 *           o MiSubclassSearch* from anywhere else in the app.
073 *     Button Clicks
074 *       o When a button is clicked, the relevant callback is called.
075 *         o Contacts: ControlButtonListenersContacts()
076 *         o Searches: ControlButtonListenersSearch()
077 *************************************************************************************************
078 */
079public class ButtonLedPair extends RelativeLayout
080{
081        //<editor-fold desc="FIELDS">
082        private MiSubclass miSubclass;
083        private ButtonSearch buttonSearch;
084        private ButtonLed  buttonLed;
085        private ControlButtonListeners ocl = null; // onClick listener
086        private String labelDefault = null; // e.g., "Search 1"
087        //</editor-fold>
088        //<editor-fold desc="CONSTRUCTORS">
089        @SuppressWarnings("unused")
090        public ButtonLedPair(Context context) {
091                super(context);
092                cinit();
093        }
094        @SuppressWarnings("unused")
095        public ButtonLedPair(Context context, AttributeSet attrs) {
096                super(context, attrs);
097                cinit();
098        }
099        @SuppressWarnings("unused")
100        public ButtonLedPair(Context context, AttributeSet attrs, int defStyleAttr) {
101                super(context, attrs, defStyleAttr);
102                cinit();
103        }
104        //</editor-fold>
105        //<editor-fold desc="GETTERS AND SETTERS">
106        private void setMiSubclass(MiSubclass misc) {
107                miSubclass = misc;
108        }
109        public MiSubclass getMiSubclass() { return miSubclass; }
110        private ButtonLed getButtonLed() { return buttonLed; }
111        public ButtonSearch getButtonSearch() { return buttonSearch; }
112        private ControlButtonListeners getOnClickListener() { return ocl; }
113        public void setLed(ButtonLed.Cmd cmd) {
114                buttonLed.setLed(cmd);
115        }
116        public String getLabelDefault() {
117                return labelDefault;
118        }
119        //</editor-fold>
120        //<editor-fold desc="EARLY INITIALIZATION">
121        /*
122         * Inflate everything and initialize the search buttons (label, callbacks,etc.)
123         */
124        private void cinit() {
125                LayoutInflater li = (LayoutInflater)
126                        getContext().getSystemService(Context.LAYOUT_INFLATER_SERVICE);
127                li.inflate(R.layout.pair_searchbutton, this, true);
128                buttonSearch = (ButtonSearch) findViewById(R.id.search_button);
129                buttonLed = (ButtonLed) findViewById(R.id.led);
130                configure();
131        }
132        private void configure() {
133                /*
134                 * Necessary for Android Studio's preview mode to work when displaying
135                 * this compound control.
136                 */
137                if (isInEditMode())
138                        return;
139
140                int rid = getId();
141                switch (rid) {
142                        case R.id.psb0: labelDefault = "Contacts"; ocl = new ControlButtonListenersContacts(); break;
143                        case R.id.psb1: labelDefault = "Search 1"; ocl = new ControlButtonListenersSearch(); break;
144                        case R.id.psb2: labelDefault = "Search 2"; ocl = new ControlButtonListenersSearch(); break;
145                        case R.id.psb3: labelDefault = "Search 3"; ocl = new ControlButtonListenersSearch(); break;
146                        case R.id.psb4: labelDefault = "Search 4"; ocl = new ControlButtonListenersSearch(); break;
147                        case R.id.psb5: labelDefault = "Search 5"; ocl = new ControlButtonListenersSearch(); break;
148                        case R.id.psb6: labelDefault = "Search 6"; ocl = new ControlButtonListenersSearch(); break;
149                        case R.id.psb7: labelDefault = "Search 7"; ocl = new ControlButtonListenersSearch(); break;
150                }
151                configButton(labelDefault, ocl);
152        }
153        private void configButton(String label, ControlButtonListeners ocl) {
154                buttonSearch.setText(label);
155                buttonSearch.setOnClickListener(ocl);
156                buttonSearch.setOnLongClickListener(ocl);
157                buttonSearch.setButtonLedPair(this);
158        }
159        //</editor-fold>
160        //<editor-fold desc="LATE INITIALIZATION">
161        /*
162         * Hook all the components together.
163         */
164        static public void lateinit(Activity a) {
165                // Have to delay this initialization until the MiSubclass* instances have been created.
166                initInstance(a, R.id.psb0, MiSubclassContacts.i());
167                initInstance(a, R.id.psb1, MiSubclassSearch1.i());
168                initInstance(a, R.id.psb2, MiSubclassSearch2.i());
169                initInstance(a, R.id.psb3, MiSubclassSearch3.i());
170                initInstance(a, R.id.psb4, MiSubclassSearch4.i());
171                initInstance(a, R.id.psb5, MiSubclassSearch5.i());
172                initInstance(a, R.id.psb6, MiSubclassSearch6.i());
173                initInstance(a, R.id.psb7, MiSubclassSearch7.i());
174        }
175        static private void initInstance(Activity a, int psbRid, MiSubclass miSubclass) {
176                // Hook up the two components (Button and Led) of a PairSearchButton compound control.
177                ButtonLedPair psb = (ButtonLedPair) a.findViewById(psbRid);
178                psb.setMiSubclass(miSubclass);
179                ButtonLed buttonLed = psb.getButtonLed();
180                ControlButtonListeners ocl = psb.getOnClickListener();
181                buttonLed.init(psb);
182                ocl.init(psb);
183
184                // The miSubclass will always be either MiSubclassContacts or MiSubclassSearch.
185                MiSubclassLeftPanel leftPanelSubclass = (MiSubclassLeftPanel) miSubclass;
186                leftPanelSubclass.init(psb);
187                leftPanelSubclass.setText(); // If a search was already underway, then overwrite the label.
188        }
189        //</editor-fold>
190}
191//</editor-fold>
