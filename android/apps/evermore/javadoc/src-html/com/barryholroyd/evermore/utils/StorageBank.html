013import com.barryholroyd.evermore.Configure;
014import com.barryholroyd.evermore.debugging.printstates.FragmentPrintStates;
015import com.barryholroyd.evermore.mapitems.mi_subclass.MiSubclass;
016import java.util.HashMap;
017
018//<editor-fold desc="BLOCK COMMENT">
019/*
020 * This is a basic StorageBank that retains simple values across
021 *   Activity instantiations by storing them in static variables.
022 * An earlier version, checked into Git, is fragment-based and
023 *   can handle complex objects, but at this point that introduces
024 *   unnecessary complexity and so I replaced it with this much
025 *   simpler version.
026 */
027//</editor-fold>
028public class StorageBank extends FragmentPrintStates
029{
030        //<editor-fold desc="FIELDS">
031        // This class stores instances of the following fields.
032
033        static private boolean displayRadius = Configure.DEFAULT_DISPLAY_RADIUS;
034        static private int mocklocSpinnerPosition      = 0;
035        static private final HashMap<MiSubclass,Boolean> leds = new HashMap<>(); // true = "on"
036
037        //</editor-fold>
038        //<editor-fold desc="GETTERS AND SETTERS">
039
040        static public boolean getDisplayRadius() { return displayRadius; }
041        static public void setDisplayRadius(boolean b) { displayRadius = b; }
042        static public void setLed(MiSubclass miSubclass, boolean b) { leds.put(miSubclass,b); }
043        static public boolean getLed(MiSubclass miSubclass) {
044                if (!leds.containsKey(miSubclass))
045                        leds.put(miSubclass, false); // leds are initially off
046                return leds.get(miSubclass);
047        }
048        public static int getMocklocSpinnerPosition() { return mocklocSpinnerPosition; }
049        public static void setMocklocSpinnerPosition(int mocklocSpinnerPosition) {
050                StorageBank.mocklocSpinnerPosition = mocklocSpinnerPosition;
051        }
052//</editor-fold>
053}
