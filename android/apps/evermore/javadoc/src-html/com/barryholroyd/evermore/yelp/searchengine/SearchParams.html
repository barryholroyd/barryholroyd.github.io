013public class SearchParams
014{
015        // fake category (first spot in each spinner pulldown
016        static final public String SELECTION_NONE = "--";
017
018        private final String search;
019        private String rating;
020        private final String category;
021        private final String radius; // radius is a Double (in string form)
022
023        public SearchParams(String _search, String _rating, String _category, String _radius) {
024                search = _search.equals("")     ? null : _search;
025                rating = _rating.equals("")     ? null : _rating;
026                category = _category.equals("") ? null : _category;
027                radius   = _radius;
028        }
029        public String getSearch() {
030                return search;
031        }
032        private String getRatingString() {
033                if (rating == null || rating.equals(SELECTION_NONE))
034                        rating = null;
035                return rating;
036        }
037        public boolean isRatingValid() {
038                return getRatingString() != null;
039        }
040        public Double getRatingDouble() {
041                if (!isRatingValid()) return 0.0;
042                else                  return Double.valueOf(rating);
043        }
044        public String getCategory() {
045                return category;
046        }
047        public String getRadius() {
048                return radius;
049        }
050}
