013import android.view.View;
014
015import com.barryholroyd.evermore.mapping.EmMap;
016import com.barryholroyd.evermore.yelp.YelpCommander;
017import com.barryholroyd.evermore.mapitems.mi_class.MiClassTargets;
018import com.barryholroyd.evermore.mapitems.mi_subclass.MiSubclassSearch;
019import com.barryholroyd.evermore.utils.Toaster;
020
021/*
022 * General class for handling clicks on the Search buttons in the left panel.
023 */
024public class ControlButtonListenersSearch extends ControlButtonListeners
025{
026        public void onClick(View v) {
027                if (!EmMap.isMapReady(true))
028                        return;
029                if (!MiClassTargets.targetMapItemsExist()) {
030                        Toaster.toastLong("Please place at least one Target on the map first.");
031                        return;
032                }
033                /*
034                 * When additional search types are implemented, the user should be prompted
035                 * here for which one they want to use. For now, "Yelp" is the only option.
036                 *
037                 * THIS IS WHERE YELP SEARCHES ARE INITIATED FROM.
038                 */
039                MiSubclassSearch.SearchType searchType = MiSubclassSearch.SearchType.Yelp;
040                switch (searchType) {
041                        case Yelp: YelpCommander.start(buttonLedPair); break;
042                }
043                super.onClick(v);
044        }
045        @Override
046        public boolean onLongClick(View v) {
047                if (!EmMap.isMapReady(false))
048                        return true;
049                super.onLongClick(v);
050                return true; // handled here
051        }
052}
