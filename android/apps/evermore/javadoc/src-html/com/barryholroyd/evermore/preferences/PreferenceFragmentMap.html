013import android.os.Bundle;
014import android.preference.CheckBoxPreference;
015import android.preference.ListPreference;
016import android.preference.PreferenceFragment;
017import android.view.View;
018
019import com.barryholroyd.evermore.R;
020import com.barryholroyd.evermore.settings.Settings;
021
022/**
023 * Created by Barry on 5/11/2015.
024 *
025 * Instantiated when you first click on the preference.
026 */
027@SuppressWarnings("unused")
028public class PreferenceFragmentMap extends PreferenceFragment
029{
030        @Override
031        public void onCreate(Bundle savedInstanceState) {
032                super.onCreate(savedInstanceState);
033                addPreferencesFromResource(R.xml.preference_settings_map);
034        }
035
036        @Override
037        public void onViewCreated(View view, Bundle savedInstanceState) {
038                /*
039                 * Set preferences to display as they were set in a previous
040                 * instance of the app. This is only necessary on the app's
041                 * initial start up, after installation. For some reason, the
042                 * system doesn't seem to set the preference values when the
043                 * app first starts up.
044                 */
045                ListPreference lp = (ListPreference) this.findPreference("MAP_TYPE");
046                String mapType = Integer.toString(Settings.getMapType());
047                lp.setValue(mapType);
048
049                CheckBoxPreference cb = (CheckBoxPreference) this.findPreference("MAP_TRAFFIC");
050                boolean mapTraffic = Settings.getMapTraffic();
051                cb.setChecked(mapTraffic);
052        }
053}
