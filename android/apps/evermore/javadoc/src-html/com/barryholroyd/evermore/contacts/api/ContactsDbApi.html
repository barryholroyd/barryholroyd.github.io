027/*
028 *  --------------------------------------------------------------------------------------------
029 *  ARCHITECTURE
030 *  --------------------------------------------------------------------------------------------
031 *  There are three modules in the Contacts subsystem.
032 *
033 *    1. Service
034 *    2. Database
035 *    3. API (this module).
036 *
037 *  The purpose of this API module is provide an API to the Contacts subsystem
038 *  that the rest of the system can use.
039 *
040 *  Both ContactsDbApi and ContactHow are exposed to the rest of the system.
041 *    They both provide access to Contacts (e.g., getAllContacts()).
042 *    ContactHow provides an interface to Home-Or-Work specific entities (e.g.,
043 *    getHowContactMapItemsWithinRadius()).
044 *
045 *  TBD: this could probably be cleaned up a bit so that ContactDbApi only provides
046 *  access to Contacts and the ContactHow provides only access to ContactHows.
047 *
048 *  Please see https://github.com/barryholroyd/TapIt/wiki/Architecture#contacts
049 *  for details as well as comments below.
050 */
051public class ContactsDbApi
052{
053        private static final BhLogger bhrel = BhLogger.bhrel();
054        static private ContactsDb contactsDb = null;
055

062        static public int getAllContactsCount() {

069        static private ContactsDb getContactsDb() {
070                if (contactsDb == null)
071                        throw new IllegalStateException("Contacts Db instance not initialized yet.");
072                return contactsDb;
073        }
074        static public ContactsHs getAllContactsByHow(AdrsTypeEnum how) {
075                ContactsHs contactsHs = new ContactsHs();
076
077                String keyLlStatus    = null;
078                switch (how) {
079                        case HOME: keyLlStatus = ContactsDb.KEY_HOME_LATLNG_STATUS; break;
080                        case WORK: keyLlStatus = ContactsDb.KEY_WORK_LATLNG_STATUS; break;
081                }
082                String where = String.format("%s = %s", keyLlStatus, LatLngStatus.VALID.toInt());
083
084                Cursor c = ContactsDb.getContactsFiltered(where);
085                if (c != null) {
086                        while (c.moveToNext())
087                                contactsHs.add(new Contact(c));
088                        c.close();
089                }
090                return contactsHs;
091        }
092        static public ContactsHs getAllContacts() {
093                ContactsHs contactsHs = new ContactsHs();
094                Cursor c = ContactsDb.getContactsFiltered(null);
095                if (c != null) {
096                        while (c.moveToNext())
097                                contactsHs.add(new Contact(c));
098                        c.close();
099                }
100                return contactsHs;
101        }
102        static public ContactsHs getHowContactsWithinRadius(
103                AdrsTypeEnum how, double target_lat, double target_lng, double radius) {
104                /*
105                 * For a given Contact address (latitude, longitude), check to
106                 *   see if it is within the specified radius of the given
107                 *   target (latitude, longitude). If it is, create a Contact
108                 *   for it and add it to the contactsHs HashMap.
109                 *
110                 * This method is called separately for Home and Work
111                 *   addresses and if it creates a new contactsHs instance it marks
112                 *   the instance appropriately.
113                 *
114                 * Checking the distance is actually a two-step activity. We first
115                 *   define a box using EmMapDistance and retrieve all the Contacts
116                 *   within that box. Then, for each of those Contacts, we refine the
117                 *   distance check using isAddressWithinRadius().
118                 */
119                String keyLlStatus    = null;
120                String keyLat          = null;
121                String keyLng          = null;
122                switch (how) {
123                        case HOME:
124                                keyLlStatus   = ContactsDb.KEY_HOME_LATLNG_STATUS;
125                                keyLat         = ContactsDb.KEY_HOME_ADRS_LAT;
126                                keyLng         = ContactsDb.KEY_HOME_ADRS_LNG;
127                                break;
128                        case WORK:
129                                keyLlStatus   = ContactsDb.KEY_WORK_LATLNG_STATUS;
130                                keyLat         = ContactsDb.KEY_WORK_ADRS_LAT;
131                                keyLng         = ContactsDb.KEY_WORK_ADRS_LNG;
132                                break;
133                }
134
135                String s =
136                        "("                                         +
137                         "%s = %s AND "                     +
138                         "%s >= %f AND %s <= %f"        +
139                         " AND "                                +
140                         "%s >= %f AND %s <= %f"        +
141                         ")";
142
143                String where = String.format(s,
144                        keyLlStatus, LatLngStatus.VALID.toInt(),
145                        keyLat, (target_lat - (EmMapDistance.max_lat) * Settings.getRadiusDistance()),
146                        keyLat, (target_lat + (EmMapDistance.max_lat) * Settings.getRadiusDistance()),
147                        keyLng, (target_lng - (EmMapDistance.max_lng) * Settings.getRadiusDistance()),
148                        keyLng, (target_lng + (EmMapDistance.max_lng) * Settings.getRadiusDistance()));
149
150                Cursor c = ContactsDb.getContactsFiltered(where);
151
152                // We now have a (hopefully small) superset of the desired set of Contacts.
153                ContactsHs contactsHs = new ContactsHs();
154                if (c != null) {
155                        while (c.moveToNext()) {
156                                double contact_lat = ContactsDbUtils.getDbDouble(c, keyLat);
157                                double contact_lng = ContactsDbUtils.getDbDouble(c, keyLng);
158                                EmLatLng tillContact = new EmLatLng(contact_lat, contact_lng);
159                                EmLatLng tillTarget = new EmLatLng(target_lat, target_lng);
160                                if (isAddressWithinRadius(tillTarget, tillContact, radius)) {
161                                        contactsHs.add(new Contact(c));
162                                }
163                        }
164                        c.close();
165                }
166                return contactsHs;
167        }
168
169        static private boolean isAddressWithinRadius(
170                EmLatLng tillTarget, EmLatLng tillContact, double radius
171        ) {
172                // Is this address within the DEFAULT_RADIUS centered at Target?
173                return EmMapDistance.isWithinRadius(radius, tillTarget, tillContact);
174        }
175        static public Contact getContact(long id) {
176                // Return a single contact.
177                /*
178                 * The id should have been obtained from a Contact instance
179                 * obtained earlier; it is the unique ID used by the Contacts
180                 * Content Provider.
181                 */
182                String where = String.format("%s=%d", ContactsDb.KEY_CONTACT_ID, id);
183                Cursor c = null;
184                try {
185                        c = ContactsDb.getContactsFiltered(where);
186                        if (c == null || c.getCount() == 0)
187                                return null;
188                        c.moveToNext();
189                        return (new Contact(c));
190                }
191                catch (Exception e) {
192                        bhrel.exception(e);
193                        throw e;
194                }
195                finally {
196                        if (c != null)
197                                c.close();
198                }
199        }
