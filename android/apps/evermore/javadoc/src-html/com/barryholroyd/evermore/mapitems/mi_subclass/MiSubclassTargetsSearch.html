013import com.barryholroyd.evermore.mapitems.mapitem.MapItemIcon;
014import com.barryholroyd.evermore.mapitems.mi_class.MiClass;
015import com.barryholroyd.evermore.mapitems.mapitem.MapItemIcon.Subclasses;
016
017public class MiSubclassTargetsSearch extends MiSubclassTargets {
018
019        static private MiSubclassTargetsSearch miSubclassTargetsSearch = null;
020
021        private MiSubclassTargetsSearch(MiClass p) {
022                super(p);
023                this.rgbColor = MapItemIcon.getRgb(Subclasses.TargetsAddress);
024                this.icon = MapItemIcon.getDrawable(Subclasses.TargetsAddress);
025                this.bmdPrimary = MapItemIcon.getBmdSearchOrTarget(Subclasses.TargetsAddress);
026        }
027
028        static public MiSubclassTargetsSearch create(MiClass p) {
029                if (miSubclassTargetsSearch == null)
030                        miSubclassTargetsSearch = new MiSubclassTargetsSearch(p);
031                return miSubclassTargetsSearch;
032        }
033        static public MiSubclassTargetsSearch i() { return miSubclassTargetsSearch; }
034
035}
