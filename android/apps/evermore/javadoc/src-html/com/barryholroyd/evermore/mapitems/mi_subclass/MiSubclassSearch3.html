013import com.barryholroyd.evermore.mapitems.mapitem.MapItemIcon;
014import com.barryholroyd.evermore.mapitems.mi_class.MiClass;
015import com.barryholroyd.evermore.mapitems.mapitem.MapItemIcon.Subclasses;
016import com.barryholroyd.evermore.types.SearchInfo;
017
018public class MiSubclassSearch3 extends MiSubclassSearch {
019        static private MiSubclassSearch3 miSubclassSearch3 = null;
020
021        static private SearchInfo searchInfo = null;
022
023        private MiSubclassSearch3(MiClass p) {
024                super(p);
025                this.rgbColor = MapItemIcon.getRgb(Subclasses.Search3);
026                this.icon = MapItemIcon.getDrawable(Subclasses.Search3);
027                this.bmdPrimary = MapItemIcon.getBmdSearchOrTarget(Subclasses.Search3);
028        }
029
030        static public MiSubclassSearch3 create(MiClass p) {
031                if (miSubclassSearch3 == null)
032                        miSubclassSearch3 = new MiSubclassSearch3(p);
033                return miSubclassSearch3;
034        }
035        static public MiSubclassSearch3 i() { return miSubclassSearch3; }
036
037        @Override
038        public SearchInfo getSearchInfo() { return searchInfo; }
039        @Override
040        public void setSearchInfo(SearchInfo _searchInfo) { searchInfo = _searchInfo; }
041
042}
