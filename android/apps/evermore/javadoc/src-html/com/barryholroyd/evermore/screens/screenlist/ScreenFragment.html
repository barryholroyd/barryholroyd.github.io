013import android.os.Bundle;
014import android.view.LayoutInflater;
015import android.view.View;
016import android.view.ViewGroup;
017
018import com.barryholroyd.evermore.ActivityMain;
019import com.barryholroyd.evermore.R;
020import com.barryholroyd.evermore.debugging.printstates.FragmentPrintStates;
021import com.barryholroyd.evermore.screens.ScreenStateManager;
022
023public class ScreenFragment extends FragmentPrintStates
024{
025        @Override
026        public View onCreateView(LayoutInflater inflater, ViewGroup container, Bundle
027                savedInstanceState) {
028                int layout = 0;
029                switch (ScreenStateManager.getScreenState()) {
030                        case MAIN:
031                                layout = R.layout.screen_main;
032                                break;
033                        case MAP:
034                                layout = R.layout.screen_map;
035                                break;
036                        case LIST:
037                                layout = R.layout.screen_list;
038                                break;
039                }
040                return inflater.inflate(layout, container, false);
041        }
042        @Override
043        public void onViewCreated(View v, Bundle savedInstanceState) {
044                /*
045                 * After creating a fragment to inflate one of the three screen types
046                 *   (Main, Map, List), this is our first chance to initialize them
047                 *   (before this point, the Views haven't been inflated and attached
048                 *   to the view hierarchy).
049                 *
050                 * initScreen() initializes the system according to the selected screen.
051                 *   As part of that process, if the screen is either Main or Map, it
052                 *   creates a EmMap instance which, in turn, creates a Google Map
053                 *   instance. Only after that has been completed can we run any code
054                 *   that depends on the existence of the Google Map instance.
055                 */
056                // Init various View callbacks, etc.
057                ActivityMain.getScreenStateManager().initScreen();
058        }
059}
