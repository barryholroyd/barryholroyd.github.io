013import com.barryholroyd.evermore.ActivityMain;
014import com.barryholroyd.evermore.dialogs.DeleteTargetDf;
015import com.barryholroyd.evermore.mapitems.mapitem.*;
016import com.barryholroyd.evermore.mapitems.mi_subclass.*;
017import com.barryholroyd.evermore.mapitems.mi_type.MiType;
018import com.google.android.gms.maps.model.Marker;
019
020public class MiClassTargets extends MiClass {
021        private static MiClassTargets miClassTargets = null;
022
023        private MiClassTargets(MiType p) {
024                this.parent = p;
025                addMiSubclass(MiSubclassTargetsContacts.create(this));
026                addMiSubclass(MiSubclassTargetsManual.create(this));
027                addMiSubclass(MiSubclassTargetsSearch.create(this));
028        }
029
030        static public MiClassTargets create(MiType p) {
031                if (miClassTargets == null)
032                        miClassTargets = new MiClassTargets(p);
033                return miClassTargets;
034        }
035        static public MiClassTargets i() { return miClassTargets; }
036
037        @SuppressWarnings("BooleanMethodIsAlwaysInverted")
038        static public boolean targetMapItemsExist() {
039                for (MiSubclass miSubclass : i().getMiSubclasses()) {
040                        if (MapItemRegistry.hasMapItems(miSubclass))
041                                return true;
042                }
043                return false;
044        }
045        static public void displayRadiusForAllTargets() {
046                for (MiSubclass miSubclass : i().getMiSubclasses()) {
047                        for (MapItem mapItem : MapItemRegistry.getMapItems(miSubclass)) {
048                                MapItemTarget mapItemTarget = (MapItemTarget) mapItem;
049                                mapItemTarget.displayRadius();
050                        }
051                }
052        }
053        public void markerInfoWindowClicked(Marker marker) {
054                DeleteTargetDf.showMe(ActivityMain.getActivity(), marker);
055        }
056}
