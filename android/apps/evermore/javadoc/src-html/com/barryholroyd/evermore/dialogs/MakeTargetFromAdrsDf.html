035/*
036 * This allows the user to specify a target using an address or a longitude and latitude
037 * of the form (e.g.): 37.375865, -122.059747.
038 *
039 * Search are done using Android's Geocoder.
040 */
041public class MakeTargetFromAdrsDf extends DialogFragmentPrintStates
042{
043        //<editor-fold desc="PUBLIC API">
044        static public void showMe(Activity a) {
045                FragmentManager fm = a.getFragmentManager();
046                MakeTargetFromAdrsDf newFragment = new MakeTargetFromAdrsDf();
047                newFragment.show(fm, "searchTargetDf");
048        }
049        //</editor-fold>
050        //<editor-fold desc="OVERRIDES">
051        @Override
052        public void onAttach(Activity activity) {
053                extendLogtag("TRGT_ADRS");
054                super.onAttach(activity);
055        }
056        @Override
057        public Dialog onCreateDialog(Bundle savedInstanceState) {
058                /*
059                 * From: http://developer.android.com/guide/topics/ui/dialogs.html.
060                 *   The system calls this only when creating the layout in a dialog.
061                 *   The only reason you might override this method when using onCreateView()
062                 *   is to modify any dialog characteristics.
063                 * This includes modifying its Window's characteristics.
064                 */
065                Dialog dialog = super.onCreateDialog(savedInstanceState);
066                UI.configureDialogWindow(dialog);
067                return dialog;
068        }
069        @Override
070        public View onCreateView(LayoutInflater inflater, ViewGroup container, Bundle savedInstanceState) {
071                View v = inflater.inflate(R.layout.search_target, container, false);
072
073                Button buttonSearch = (Button) v.findViewById(R.id.target_button_search);
074                configButtonSearch(buttonSearch, this);
075
076                Button buttonDone = (Button) v.findViewById(R.id.target_button_cancel);
077                UI.configureDialogButtonDone(buttonDone, this, true);
078
079                return v;
080        }
081        //</editor-fold>
082        //<editor-fold desc="CONFIGURATION">
083        private void configButtonSearch(Button buttonSearch, final DialogFragment df) {
084                buttonSearch.setOnClickListener(new OnClickSearchTarget(df));
085        }
086        //</editor-fold>
087        class OnClickSearchTarget implements View.OnClickListener {
088                DialogFragment df = null;
089                public OnClickSearchTarget(DialogFragment dialogFragment) {
090                        df = dialogFragment;
091                }
092                public void onClick(View v) {
093                        String adrs = getValue(v, R.id.search_target_edit);
094
095                        EmLatLng till = EmLatLngResolver.getLatLng(adrs, true);
096                        if (till.getStatus() == LatLngStatus.VALID) {
097                                dropTarget(adrs, till);
098                                Support.hideKeyboard(v);
099                                df.dismiss();
100                                EmPosition tip = new EmPosition(till);
101                                EmMap.getInstance().moveToLocation(tip, false, true, false);
102                        }
103                }
104                private void dropTarget(String adrs, EmLatLng till) {
105                        if (!EmMap.isMapReady(true))
106                                return;
107                        @SuppressWarnings("UnnecessaryLocalVariable")
108                        String title = adrs;
109                        MiSubclassTargetsSearch.i().createMapItem(title, null, till.getLat(), till.getLng());
110                }
111                private String getValue(View v, int rid) {
112                        ViewGroup grandparent   = (ViewGroup) v.getParent().getParent();
113                        TextView  tv       = (TextView) grandparent.findViewById(rid);
114                        return tv.getText().toString();
115                }
116        }
117}
