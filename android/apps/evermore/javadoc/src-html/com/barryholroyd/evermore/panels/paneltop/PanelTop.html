013import android.app.Activity;
014import android.view.View;
015import android.widget.Button;
016import android.widget.LinearLayout;
017
018import com.barryholroyd.evermore.R;
019import com.barryholroyd.evermore.dialogs.MakeTargetFromContactDf;
020import com.barryholroyd.evermore.dialogs.MakeTargetFromAdrsDf;
021import com.barryholroyd.evermore.mapping.EmMap;
022
023public class PanelTop
024{
025
026        static public void init(Activity a) {
027                LinearLayout ll = (LinearLayout) a.findViewById(R.id.bar_top);
028                setButton(ll, R.id.button_contact_target, new ContactTargetButton(a));
029                setButton(ll, R.id.button_search_target,  new SearchTargetButton(a));
030        }
031        static private void setButton(LinearLayout ll, int rId, View.OnClickListener cback) {
032                Button b = (Button) ll.findViewById(rId);
033                b.setOnClickListener(cback);
034        }
035}
036class ContactTargetButton implements View.OnClickListener {
037        private Activity a = null;
038        ContactTargetButton(Activity activity) { a = activity; }
039        public void onClick(View v) {
040                if (!EmMap.isMapReady(true))
041                        return;
042                MakeTargetFromContactDf.showMe(a);
043        }
044}
045class SearchTargetButton implements View.OnClickListener {
046        private Activity a = null;
047        SearchTargetButton(Activity activity) { a = activity; }
048        public void onClick(View v) {
049                if (!EmMap.isMapReady(true))
050                        return;
051                MakeTargetFromAdrsDf.showMe(a);
052        }
053}
