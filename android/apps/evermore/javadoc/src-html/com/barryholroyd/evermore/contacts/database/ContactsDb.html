070        public ContactsDb(Context context) {
071                if (dbHelper == null) {
072                        dbHelper = new DbHelper(context,
073                                DbHelper.DATABASE_NAME, null,
074                                DbHelper.DATABASE_VERSION);
075                }
076        }

079        /*
080         * These methods are public, but only accessed within the contacts module.
081         */
082        @Nullable
083        static public Cursor getContactsFiltered(String where) {
084                SQLiteDatabase db       = dbHelper.getReadableDatabase();
085                SQLiteQueryBuilder qb   = new SQLiteQueryBuilder();
086
087                qb.setTables(DbHelper.CONTACT_ITEMS_TABLE);
088                String[] selection = {
089                        KEY_CONTACT_ID,         // Unique id from the Contacts ContentProvider.
090                        KEY_DISPLAY_NAME,
091                        KEY_HOME_ADRS,
092                        KEY_HOME_LATLNG_STATUS,
093                        KEY_HOME_ADRS_LAT,
094                        KEY_HOME_ADRS_LNG,
095                        KEY_WORK_ADRS,
096                        KEY_WORK_LATLNG_STATUS,
097                        KEY_WORK_ADRS_LAT,
098                        KEY_WORK_ADRS_LNG,
099                        KEY_EMAIL_HOME,
100                        KEY_EMAIL_MOBILE,
101                        KEY_EMAIL_WORK,
102                        KEY_EMAIL_OTHER,
103                        KEY_PHONE_HOME,
104                        KEY_PHONE_MOBILE,
105                        KEY_PHONE_WORK
106                };
107
108                String order_by = String.format("%s %s", KEY_CONTACT_ID, "ASC");
109                String where_args[] = null;
110                String group_by = null;
111                String having = null;
112
113                Cursor c;
114                try {
115                        //noinspection ConstantConditions
116                        c = qb.query(db, selection, where, where_args, group_by, having, order_by);
117                } catch (Exception e) {
118                        bhrel.exception(e);
119                        throw e;
120                }
121                return c;
122        }
123        static public int getContactsAllCount() {
124                // Return a count of the total number of contacts in the database.
125                SQLiteDatabase db       = dbHelper.getReadableDatabase();
126                String select = String.format("SELECT COUNT(*) FROM %s", DbHelper.CONTACT_ITEMS_TABLE);
127                Cursor c = db. rawQuery(select, null);
128                c.moveToFirst();
129                int cnt = c.getInt(0);
130                c.close();
131                return cnt;
132        }
133        public void insertRecord(Contact contact) {
134                Contact.trace("INS", contact);
135                ContentValues values = fillContentValues(contact);
136                SQLiteDatabase db = dbHelper.getWritableDatabase();
137                db.insert(DbHelper.CONTACT_ITEMS_TABLE, null, values);
138        }
139        public void updateRecord(Contact contact){
140                Contact.trace("UPD", contact);
141                SQLiteDatabase db = dbHelper.getWritableDatabase();
142                ContentValues values = fillContentValues(contact);
143                String where = KEY_CONTACT_ID + "=" + contact.getId();
144                db.update(DbHelper.CONTACT_ITEMS_TABLE, values, where, null);
145        }
146        public void deleteRecord(Contact contact) {
147                Contact.trace("DEL", contact);
148                String where = KEY_CONTACT_ID + "=" + contact.getId();
149                deleteRecords(where);
150        }
151        public void clearRecords() {
152                Contact.trace("DEL", "Deleting all contacts from the database.");
153                deleteRecords("");
154        }
155        private void deleteRecords(String where) {
156                SQLiteDatabase db = dbHelper.getWritableDatabase();
157                db.delete(DbHelper.CONTACT_ITEMS_TABLE, where, null);
158        }

160        //<editor-fold desc="PRIVATE METHODS"/>
161        private static class DbHelper extends SQLiteOpenHelper  {
162                /*
163                 * SQLiteOpenHelper implements getReadableDatabase() and getWritableDatabase().
164                 * When either of those ic called, it will instantiate the database if it
165                 * doesn't already exist. It will also call onCreate() so that the overridden
166                 * version of that (below) can add the necessary table(s). onUpgrade() is also
167                 * called, if/as appropriate.
168                 */
169                private static final String DATABASE_NAME = "evermore.db";
170                private static final String CONTACT_ITEMS_TABLE = "contact_items";
171                private static final int DATABASE_VERSION = 7;
172
173                private static final String CREATE_CONTACT_ITEMS_TABLE =
174                        "CREATE TABLE " + CONTACT_ITEMS_TABLE + " ("
175                                + KEY_CONTACT_ID                 + " INTEGER PRIMARY KEY, "  // contacts primary key
176                                + KEY_DISPLAY_NAME               + " TEXT, "
177                                + KEY_HOME_ADRS               + " TEXT, "
178                                + KEY_HOME_LATLNG_STATUS    + " INTEGER, "
179                                + KEY_HOME_ADRS_LAT           + " DOUBLE, "
180                                + KEY_HOME_ADRS_LNG           + " DOUBLE, "
181                                + KEY_WORK_ADRS               + " TEXT, "
182                                + KEY_WORK_LATLNG_STATUS    + " INTEGER, "
183                                + KEY_WORK_ADRS_LAT           + " DOUBLE, "
184                                + KEY_WORK_ADRS_LNG           + " DOUBLE, "
185                                + KEY_EMAIL_HOME                 + " TEXT,"
186                                + KEY_EMAIL_MOBILE               + " TEXT,"
187                                + KEY_EMAIL_WORK                 + " TEXT,"
188                                + KEY_EMAIL_OTHER                + " TEXT,"
189                                + KEY_PHONE_HOME                 + " TEXT,"
190                                + KEY_PHONE_MOBILE               + " TEXT,"
191                                + KEY_PHONE_WORK                 + " TEXT"
192                                + ");";
193
194                private static final String DROP_TABLE_LOCATIONS =
195                        "DROP TABLE " + CONTACT_ITEMS_TABLE + ";";
196
197                public DbHelper(
198                        Context context, String name, SQLiteDatabase.CursorFactory factory, int version) {
199                        super(context, name, factory, version);
200                }
201                @Override
202                public void onCreate(SQLiteDatabase db) {
203                        createDb(db);
204                }
205                @Override
206                public void onUpgrade(SQLiteDatabase db, int oldVersion, int newVersion) {
207                        bhrel.log("Upgrading database from %d to %d, which will destroy all data.",
208                                oldVersion, newVersion);
209                        deleteDb(db);
210                        createDb(db);
211                }
212                @SuppressWarnings("unused")
213                private void clearDb(SQLiteDatabase db) {
214                        db.delete(CONTACT_ITEMS_TABLE, "1", null); // Clear out rows in the relevant tables.
215                }
216                private void createDb(SQLiteDatabase db) {
217                        db.execSQL(CREATE_CONTACT_ITEMS_TABLE);    // Create the desired tables.
218                }
219                private void deleteDb(SQLiteDatabase db) {
220                        db.execSQL(DROP_TABLE_LOCATIONS);
221                }
222        }
223        private ContentValues fillContentValues(Contact contact) {
224                // Create a ContentValues for use with insertRecord() and updateRecord().
225                ContentValues values = new ContentValues();
226                values.put(KEY_CONTACT_ID, contact.getId());
227                values.put(KEY_DISPLAY_NAME, contact.getDisplayName());
228
229                values.put(KEY_HOME_ADRS, contact.getHomeAdrs().getAdrs());
230                values.put(KEY_HOME_LATLNG_STATUS, contact.getHomeLocationStatus().toInt());
231                values.put(KEY_HOME_ADRS_LAT, contact.getHomeLat());
232                values.put(KEY_HOME_ADRS_LNG, contact.getHomeLng());
233
234                values.put(KEY_WORK_ADRS, contact.getWorkAdrs().getAdrs());
235                values.put(KEY_WORK_LATLNG_STATUS, contact.getWorkLocationStatus().toInt());
236                values.put(KEY_WORK_ADRS_LAT, contact.getWorkLat());
237                values.put(KEY_WORK_ADRS_LNG, contact.getWorkLng());
238
239                values.put(KEY_EMAIL_HOME, contact.getEmailHome());
240                values.put(KEY_EMAIL_MOBILE, contact.getEmailMobile());
241                values.put(KEY_EMAIL_WORK, contact.getEmailWork());
242                values.put(KEY_EMAIL_OTHER, contact.getEmailOther());
243                values.put(KEY_PHONE_HOME, contact.getPhoneHome());
244                values.put(KEY_PHONE_MOBILE, contact.getPhoneMobile());
245                values.put(KEY_PHONE_WORK, contact.getPhoneWork());
246
247                return values;
248        }
