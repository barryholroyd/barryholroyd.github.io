013import com.barryholroyd.evermore.mapitems.mi_subclass.MiSubclass;
014import com.barryholroyd.evermore.mapping.EmMap;
015import com.barryholroyd.evermore.utils.StorageBank;
016import com.google.android.gms.maps.GoogleMap;
017import com.google.android.gms.maps.model.BitmapDescriptor;
018import com.google.android.gms.maps.model.Circle;
019import com.google.android.gms.maps.model.LatLng;
020
021public class MapItemTarget extends MapItem
022{
023        private Circle radius = null;
024        public MapItemTarget(GoogleMap gmap, MiSubclass misc, String title, String snippet,
025                             LatLng ll, BitmapDescriptor bitmap_descriptor) {
026                super(gmap, misc, title, snippet, ll, bitmap_descriptor, title, null);
027        }
028        @Override
029        public void refresh() {
030                displayRadius();
031        }
032        public void displayRadius() {
033                destroyRadius();
034                radius = EmMap.getRadius(this);
035                boolean b = StorageBank.getDisplayRadius();
036                radius.setVisible(b);
037        }
038        @Override
039        public void destroy() {
040                destroyRadius();
041                super.destroy();
042        }
043        private void destroyRadius() {
044                if (radius != null)
045                        radius.remove();
046        }
047}
